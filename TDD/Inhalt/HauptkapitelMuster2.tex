	\section{Hauptkapitel 2.2}
	
	Musteraufzählung:
			\begin{itemize} [leftmargin = 2cm, rightmargin = 2cm]
				\item[$\mathcal{X}$] Eine Menge an gewichteten Partikeln 
				\item[$w^{[k]}$] Beschreibt die Gewichtung des Partikel $k$
				\item[$\F Q$] Kovarianz der Messung
				\item[$\hat{\F z}^{[k]}$] Annahme der Messung mit $\hat{\F z}^{[k]} = h(\FM \mu_{j,t}^{[k]},\F x_t^{[k]})$
				\item[$\F c_t$] Zum Zeitpunkt $t$ beobachteter Orientierungspunkt
				\item[$\FM \mu_{n,t-1}^{[k]}$] Erwartungswertvektor des $n$-ten Orientierungspunktes von Partikel $k$
				\item[$\FM \Sigma_{n,t-1}^{[k]}$] Kovarianzmatrix des $n$-ten Orientierungspunktes von Partikel $k$
			\end{itemize}
